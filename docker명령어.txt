2. cmd 창에서 실행

#도커 모든 이미지 보기
docker images

#도커 이미지 삭제
docker rmi mysql

#도커 컨테이너 삭제
docker rm mysql-container

#이미지 다운로드
docker pull mysql

#컨테이너 생성
docker create --name mysql-container -e MYSQL_ROOT_PASSWORD=1234 -p 3306:3306 -v ./mysql:/var/lib/mysql
docker create --name mysql-container -e MYSQL_ROOT_PASSWORD=1234 -p 3306:3306 mysql

#컨테이너 시작
docker start -d mysql-container

# mysql docker 컨테이너 실행 및 이미지 다운로드
docker run --name mysql-container -e MYSQL_ROOT_PASSWORD=1234 -d -p 3306:3306 mysql:latest
